def is_likely_exchange(address, tx_count):
    """Filter out exchange wallets"""
    if address.lower() in KNOWN_EXCHANGE_ADDRESSES:
        return True
    # Heuristic: exchanges have insane tx counts
    if tx_count > 100000:
        return True
    return False
